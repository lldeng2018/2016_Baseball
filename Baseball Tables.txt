CREATE TABLE TEAM (
    Team VARCHAR(50) NOT NULL,
    TeamID VARCHAR(5) NOT NULL,
    CONSTRAINT TEAM_PK PRIMARY KEY (TeamID)
);

CREATE TABLE VENUE (
    Venue VARCHAR (50) NOT NULL,
    VenueID VARCHAR (5) NOT NULL,
    FieldType CHAR (10),
    IndoorOutdoor CHAR (10),
    CONSTRAINT VENUE_PK PRIMARY KEY (VenueID) 
);
    
CREATE TABLE DETAIL (
    GameID VARCHAR (5) NOT NULL,
    TeamID VARCHAR (5) NOT NULL,
    Team VARCHAR (50) NOT NULL,
    Errors NUMBER (5),
    Hits NUMBER (5),
    Runs NUMBER (5),
    Outome CHAR (5),
    HomeAway CHAR (5),
CONSTRAINT GAME_FK_GAME FOREIGN KEY (GameID) REFERENCES GAME (GAME.GameID)
   ON DELETE CASCADE,
CONSTRAINT TeamID_FK_TEAM FOREIGN KEY (TeamID) REFERENCES TEAM (GAME.TeamID)
   ON DELETE SET NULL);

CREATE TABLE GAME (
    GAMEID VARCHAR (5) NOT NULL,
    GameDate DATE,
    Attendence NUMBER (5),
    WindSpeed NUMBER (5),
    WindDirection VARCHAR (50),
    Temperature NUMBER (5),
    VenueID VARCHAR (5),
CONSTRAINT GAME_PK PRIMARY KEY (GAMEID),  
CONSTRAINT GAME_FK FOREIGN KEY (VenueID) REFERENCES VENUE (VenueID));
